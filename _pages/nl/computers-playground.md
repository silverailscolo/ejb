<script>
  function showTheme() {
   var result = determineComputedTheme();
   var elm = document.getElementById('mydiv');
   var d = document.createElement('div');
   d.innerText = result;
   elm.appendChild(d);
  }
  document.addEventListener("DOMContentLoaded", showTheme);
